# pytr App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a pytr app that downloads Trade Republic documents into paperless-ngx, with an HTTP-based OTP login flow triggered manually via ArgoCD.

**Architecture:** Two suspended/scheduled CronJobs sharing a credentials PVC. The login CronJob runs a Python wrapper that starts `pytr login` as a subprocess and exposes an HTTP endpoint for OTP code submission. The download CronJob runs `pytr dl_docs` on a monthly schedule. A Traefik IngressRoute (behind oauth2-proxy) routes to the login Job's HTTP server.

**Tech Stack:** Helm chart (generic chart dependency), Python 3 stdlib (`http.server`, `subprocess`), pytr 0.4.7, Traefik IngressRoute, External Secrets (Akeyless)

### Task 2: Create the Python login wrapper script ConfigMap

**Files:**
- Create: `apps/pytr/templates/configmap-login-script.yaml`

- [ ] **Step 1: Create configmap-login-script.yaml**

Create `apps/pytr/templates/configmap-login-script.yaml`:

```yaml
{{- $appName := required "appName must be set in values.yaml" .Values.appName }}
{{- if .Values.pytr }}
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: {{ $appName }}-login-script
  namespace: {{ .Release.Namespace }}
data:
  login.py: |
    #!/usr/bin/env python3
    """
    pytr login wrapper with HTTP OTP code endpoint.

    Starts pytr login as a subprocess, then serves an HTTP endpoint
    that accepts the OTP code and feeds it to pytr's stdin.

    Usage: python3 login.py
    Env vars: PHONE_NUMBER, PIN
    Endpoint: GET /code/<4-digit-code>
    """
    import os
    import sys
    import subprocess
    import threading
    from http.server import HTTPServer, BaseHTTPRequestHandler

    LOGIN_PROCESS = None
    LOGIN_RESULT = None


    class CodeHandler(BaseHTTPRequestHandler):
        def do_GET(self):
            global LOGIN_PROCESS, LOGIN_RESULT

            # Parse /code/XXXX
            parts = self.path.strip("/").split("/")
            if len(parts) == 2 and parts[0] == "code" and parts[1].isdigit() and len(parts[1]) == 4:
                code = parts[1]

                if LOGIN_PROCESS is None or LOGIN_PROCESS.poll() is not None:
                    self.send_response(500)
                    self.send_header("Content-Type", "text/plain")
                    self.end_headers()
                    self.wfile.write(b"Login process is not running.\n")
                    return

                try:
                    LOGIN_PROCESS.stdin.write(code.encode() + b"\n")
                    LOGIN_PROCESS.stdin.flush()

                    # Wait for pytr to finish (timeout 60s)
                    LOGIN_PROCESS.wait(timeout=60)
                    rc = LOGIN_PROCESS.returncode

                    if rc == 0:
                        self.send_response(200)
                        self.send_header("Content-Type", "text/plain")
                        self.end_headers()
                        self.wfile.write(b"Login successful!\n")
                        LOGIN_RESULT = "success"
                    else:
                        self.send_response(500)
                        self.send_header("Content-Type", "text/plain")
                        self.end_headers()
                        self.wfile.write(f"Login failed with exit code {rc}.\n".encode())
                        LOGIN_RESULT = "failure"
                except subprocess.TimeoutExpired:
                    self.send_response(500)
                    self.send_header("Content-Type", "text/plain")
                    self.end_headers()
                    self.wfile.write(b"Login timed out.\n")
                    LOGIN_RESULT = "failure"
                except Exception as e:
                    self.send_response(500)
                    self.send_header("Content-Type", "text/plain")
                    self.end_headers()
                    self.wfile.write(f"Error: {e}\n".encode())
                    LOGIN_RESULT = "failure"
            elif self.path == "/health":
                self.send_response(200)
                self.send_header("Content-Type", "text/plain")
                self.end_headers()
                running = LOGIN_PROCESS is not None and LOGIN_PROCESS.poll() is None
                self.wfile.write(f"Login process running: {running}\n".encode())
            else:
                self.send_response(404)
                self.send_header("Content-Type", "text/plain")
                self.end_headers()
                self.wfile.write(b"Use /code/XXXX to submit your 4-digit code.\n")

        def log_message(self, format, *args):
            print(f"[HTTP] {format % args}", flush=True)


    def main():
        global LOGIN_PROCESS, LOGIN_RESULT

        phone = os.environ.get("PHONE_NUMBER")
        pin = os.environ.get("PIN")

        if not phone or not pin:
            print("ERROR: PHONE_NUMBER and PIN environment variables must be set", flush=True)
            sys.exit(1)

        pytr_home = os.environ.get("PYTR_HOME", "/home/pytr/.pytr")
        os.makedirs(pytr_home, exist_ok=True)
        os.environ["HOME"] = os.path.dirname(pytr_home)

        print(f"Starting pytr login for {phone}...", flush=True)

        LOGIN_PROCESS = subprocess.Popen(
            ["pytr", "login", "--phone_no", phone, "--pin", pin],
            stdin=subprocess.PIPE,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
        )

        # Print pytr output in a background thread
        def stream_output():
            for line in LOGIN_PROCESS.stdout:
                print(f"[pytr] {line.decode().rstrip()}", flush=True)

        output_thread = threading.Thread(target=stream_output, daemon=True)
        output_thread.start()

        # Give pytr a moment to start and send the auth request
        import time
        time.sleep(3)

        if LOGIN_PROCESS.poll() is not None:
            print(f"ERROR: pytr exited early with code {LOGIN_PROCESS.returncode}", flush=True)
            sys.exit(1)

        print("pytr login initiated. Waiting for OTP code via HTTP...", flush=True)
        print("Submit your code at: /code/XXXX", flush=True)

        server = HTTPServer(("0.0.0.0", 8080), CodeHandler)
        print("HTTP server listening on port 8080", flush=True)

        # Serve until we get a result
        while LOGIN_RESULT is None:
            server.handle_request()

        server.server_close()
        print(f"Login {LOGIN_RESULT}. Exiting.", flush=True)

        if LOGIN_RESULT == "success":
            sys.exit(0)
        else:
            sys.exit(1)


    if __name__ == "__main__":
        main()
{{- end }}
```

### Task 3: Create the login CronJob template

**Files:**
- Create: `apps/pytr/templates/login-cronjob.yaml`

- [ ] **Step 1: Create login-cronjob.yaml**

Create `apps/pytr/templates/login-cronjob.yaml`:

```yaml
{{- $appName := required "appName must be set in values.yaml" .Values.appName }}
{{- if .Values.pytr }}
---
apiVersion: batch/v1
kind: CronJob
metadata:
  name: {{ $appName }}-login
  namespace: {{ .Release.Namespace }}
  labels:
    app: {{ $appName }}
    component: login
spec:
  schedule: {{ .Values.pytr.login.schedule | quote }}
  suspend: {{ .Values.pytr.login.suspend }}
  concurrencyPolicy: Forbid
  successfulJobsHistoryLimit: 3
  failedJobsHistoryLimit: 3
  jobTemplate:
    spec:
      backoffLimit: 0
      activeDeadlineSeconds: 600
      template:
        metadata:
          labels:
            app: {{ $appName }}
            component: login
        spec:
          restartPolicy: Never
          containers:
          - name: login
            image: python:3-slim
            command: ["sh", "-c"]
            args:
              - |
                pip install --quiet pytr=={{ .Values.pytr.pytrVersion }} && \
                python3 /scripts/login.py
            ports:
            - containerPort: 8080
              name: http
            envFrom:
            - secretRef:
                name: {{ $appName }}-credentials
            volumeMounts:
            - name: login-script
              mountPath: /scripts
              readOnly: true
            - name: pytr-credentials
              mountPath: /home/pytr/.pytr
          volumes:
          - name: login-script
            configMap:
              name: {{ $appName }}-login-script
          - name: pytr-credentials
            persistentVolumeClaim:
              claimName: {{ $appName }}-credentials-pvc
{{- end }}
```
